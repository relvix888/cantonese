        const numberDisplay = document.getElementById('number-display');

        const charDisplay = document.getElementById('char-display');

        const jyutpingDisplay = document.getElementById('jyutping-display');

        const prevBtn = document.getElementById('prev-btn');

        const nextBtn = document.getElementById('next-btn');

        const cardNumberInput = document.getElementById('card-number-input');

        const goBtn = document.getElementById('go-btn');

        const randomBtn = document.getElementById('random-btn'); // New button

        const messageBox = document.getElementById('message-box');

        const messageText = document.getElementById('message-text');


        let currentCardIndex = 0;

        let isFlipped = false;

        let shuffledData = [...flashcardData]; // Start with a copy of the original data


        // Function to update the card content

        function updateCard() {

            const cardData = shuffledData[currentCardIndex];

            numberDisplay.textContent = cardData.Number;

            charDisplay.textContent = cardData.Char;

            jyutpingDisplay.textContent = cardData.jyutping;

            flashcard.classList.remove('is-flipped');

            isFlipped = false;

        }


        // Function to play the audio

        function playAudio() {

            const pronunciation = shuffledData[currentCardIndex].jyutping;

            if (pronunciation) {

                const audioUrl = `${mp3BaseUrl}${pronunciation}.mp3`;

                const audio = new Audio(audioUrl);

                audio.play().catch(error => {

                    console.error("Audio playback failed:", error);

                    showMessage(`Audio not found for ${pronunciation}.`);

                });

            } else {

                 showMessage("No pronunciation data for this character.");

            }

        }

        

        function showMessage(msg) {

            messageText.textContent = msg;

            messageBox.classList.remove('hidden');

            setTimeout(() => {

                messageBox.classList.add('hidden');

            }, 3000);

        }


        // Function to shuffle the array (Fisher-Yates algorithm)

        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

            return array;

        }


        // Event listeners

        flashcard.addEventListener('click', () => {

            if (!isFlipped) {

                playAudio();

            flashcard.classList.toggle('is-flipped');

            isFlipped = !isFlipped;

        });


        nextBtn.addEventListener('click', (e) => {

            e.stopPropagation(); // Prevents card flip

            updateCard();

        });


        prevBtn.addEventListener('click', (e) => {

            e.stopPropagation(); // Prevents card flip

            currentCardIndex = (currentCardIndex - 1 + shuffledData.length) % shuffledData.length;

            updateCard();

        });


        // Event listener for the "Go" button

        function goToCard() {

            const number = parseInt(cardNumberInput.value, 10);

            const foundIndex = shuffledData.findIndex(card => card.Number === number);

            

            if (foundIndex === -1) {

                showMessage(`Card with number ${number} not found.`);

            } else {

                currentCardIndex = foundIndex;

                updateCard();

            }

            cardNumberInput.value = ''; // Clear the input field

        }


        goBtn.addEventListener('click', goToCard);


        // Event listener for the Enter key on the input field

        cardNumberInput.addEventListener('keydown', (e) => {

            if (e.key === 'Enter') {

                goToCard();
